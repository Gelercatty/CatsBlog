# Docker

`docker run --rm --gpus all nvidia/cuda:12.2.2-runtime-ubuntu22.04 nvidia-smi`

`docker run -d --name dev-cuda122 --gpus all \
  --restart unless-stopped \
  --shm-size=8g \
  -v /home/mayanwen/fakeubuntu:/workspace \
  -w /workspace \
  nvidia/cuda:12.2.2-devel-ubuntu22.04 \
  tail -f /dev/null`
